1PL spectrum: baseline + two-peak fit

A gated-TMDC PL spectrum has a neutral exciton (X⁰) and a trion peak on a sloped CCD background. We subtract a baseline estimated from signal-free windows, then fit both peaks — each first alone inside a ±window, then globally so overlapping tails are shared. Drag the sliders to change the underlying physics and watch the decomposition.

Python — windowed Lorentzian fit with quality gate 
def lorentzian(x, amplitude, center, gamma):
    """Lorentzian peak: A * g^2 / ((x - c)^2 + g^2). FWHM = 2*gamma."""
    dx = x - center
    return amplitude * gamma * gamma / (dx * dx + gamma * gamma)

def fit_peak_windowed(wl, y, center_guess, window_nm=15.0,
                      gamma_bounds=(0.4, 15.0), min_snr=2.0):
    """Fit one Lorentzian inside +/- window_nm around center_guess.
    Returns None when the fit fails the quality gate."""
    m = np.abs(wl - center_guess) <= window_nm
    x, yy = wl[m], y[m]
    yy = yy - np.percentile(yy, 10)          # local baseline inside window
    peak = float(np.max(yy))
    p0 = [peak, x[np.argmax(yy)], 2.0]
    lo = [0.0, center_guess - window_nm, gamma_bounds[0]]
    hi = [4.0 * peak, center_guess + window_nm, gamma_bounds[1]]
    popt, pcov = curve_fit(lorentzian, x, yy, p0=p0, bounds=(lo, hi))
    snr = popt[0] / max(noise_level_mad(yy), 1e-12)   # MAD noise floor
    if snr < min_snr:
        return None                          # gap, not a garbage point
    return dict(center_nm=popt[1], fwhm_nm=2.0 * popt[2], amplitude=popt[0])

2Power dependence & saturation

Integrated PL vs excitation power on log-log axes: a power law I ∝ Pⁿ is a straight line of slope n, and saturation bends it over. We fit a Hill model I = Imax·Pⁿ/(Kⁿ+Pⁿ) and locate the onset with the derivative dI/d(log₁₀P), which peaks near the saturation power K. Powers are parsed from filenames like PL_P1p5uW_Vf0.asc.

Python — filename metadata + Hill fit + dI/dlogP 
# filename convention: p = decimal point, m = minus  ->  P1p5uW = 1.5 uW
POWER_RE = re.compile(r"P(?P<val>m?\d+p?\d*)(?P<unit>nW|uW|mW|W)")
UNIT_SCALE = {"nW": 1e-9, "uW": 1e-6, "mW": 1e-3, "W": 1.0}

def parse_power_from_name(name):
    m = POWER_RE.search(name)
    val = float(m.group("val").replace("m", "-").replace("p", "."))
    return val * UNIT_SCALE[m.group("unit")]

def hill_saturation(P, Imax, n, K, c):
    return Imax * P ** n / (K ** n + P ** n) + c

popt, pcov = curve_fit(hill_saturation, powers_w, intensity,
                       p0=[intensity.max(), 1.0, np.median(powers_w), 0.0],
                       bounds=([0, 0.3, 0, -np.inf], [np.inf, 3.0, np.inf, np.inf]))

def derivative_dlogp_nan(intensity, powers_w):
    """Centered finite difference of I against log10(P), NaN-tolerant."""
    logp = np.where(powers_w > 0, np.log10(powers_w), np.nan)
    out = np.full_like(intensity, np.nan, dtype=float)
    for i in range(len(intensity)):
        lo, hi = max(i - 1, 0), min(i + 1, len(intensity) - 1)
        dy, dx = intensity[hi] - intensity[lo], logp[hi] - logp[lo]
        if np.isfinite(dy) and np.isfinite(dx) and dx != 0:
            out[i] = dy / dx
    return out

3TRPL: IRF-convolved decay fitting

A TCSPC histogram is the true decay convolved with the instrument response function (IRF). Instead of deconvolving (noisy!), we fit the analytic convolution of exponentials with a Gaussian IRF — the exponentially-modified-Gaussian (erfc) form. Watch how the IRF width rounds the rising edge and how the two lifetimes show up as two slopes on the log axis.

Python — analytic IRF convolution (erfc form) 
from scipy.special import erfc

def _conv_term(t, A, tau, t0, w):
    """exp(-t/tau)*H(t) convolved with a Gaussian IRF of sigma w."""
    arg_exp  = (w * w - 2.0 * tau * (t - t0)) / (2.0 * tau * tau)
    arg_erfc = (w * w - tau * (t - t0)) / (np.sqrt(2.0) * w * tau)
    out = np.zeros_like(t, dtype=float)
    safe = arg_exp < 700.0                    # keep exp() finite
    out[safe] = 0.5 * A * np.exp(arg_exp[safe]) * erfc(arg_erfc[safe])
    return out

def biexp_irf(t, y0, t0, A1, tau1, A2, tau2, w):
    return y0 + _conv_term(t, A1, tau1, t0, w) + _conv_term(t, A2, tau2, t0, w)

# Poisson-weighted fit: sigma = sqrt(counts)
sigma = np.sqrt(np.clip(counts, 1, None))
popt, pcov = curve_fit(biexp_irf, t, counts, p0=p0, bounds=(lo, hi), sigma=sigma)

def tau_1e_from_peak(t, y):
    """Model-free cross-check: time from peak to peak/e (linear interp)."""
    imax = int(np.argmax(y)); tail = y[imax:] / y[imax]
    j = int(np.where(tail <= 1.0 / np.e)[0][0])
    frac = (1.0 / np.e - tail[j - 1]) / (tail[j] - tail[j - 1])
    return t[imax + j - 1] + frac * (t[imax + j] - t[imax + j - 1]) - t[imax]

4Transient absorption maps & chirp correction

In a pump-probe ΔR/R map the probe's time zero drifts with wavelength (group velocity dispersion) — the "chirp". We model t₀(λ) as a polynomial fitted to the coherent artifact and re-interpolate every wavelength column onto a common time axis. Crank the chirp up and toggle the correction: the bleach (blue) and photoinduced absorption (red) snap back to a flat onset.

chirp strength
Python — chirp correction by column re-interpolation 
def t0_of_wavelength(wl, coeffs=(-3.8e-5, 0.072, -31.6)):
    """Chirp model: polynomial arrival time t0(lambda), fitted in practice
    to the coherent artifact."""
    return np.polyval(coeffs, wl)

def chirp_correct(t, wl, data, coeffs, dt_uniform=0.1):
    # 1. resample onto a uniform time grid
    t_uni = np.arange(t[0], t[-1], dt_uniform)
    data_uni = np.empty((t_uni.size, wl.size))
    for j in range(wl.size):
        data_uni[:, j] = np.interp(t_uni, t, data[:, j])
    # 2. shift each wavelength column by its own t0 (spline via map_coordinates)
    t0 = t0_of_wavelength(wl, coeffs); t0 -= t0.mean()
    idx = (t_uni[:, None] + t0[None, :] - t_uni[0]) / dt_uniform
    coords = np.stack([idx, np.broadcast_to(np.arange(wl.size), idx.shape)])
    corrected = map_coordinates(data_uni, coords, order=1, mode="nearest")
    # 3. crop to the window valid for every column
    lo = max(0, int(np.ceil(-t0.min() / dt_uniform)))
    hi = corrected.shape[0] - max(0, int(np.ceil(t0.max() / dt_uniform)))
    return t_uni[lo:hi], corrected[lo:hi]

5Gate-dependent PL: species tracking

Sweeping the gate voltage moves the Fermi level: at charge neutrality the neutral exciton X⁰ dominates; with doping, oscillator strength transfers to the trion, which also red-shifts. Each row of the map is one spectrum; a windowed double-Gaussian fit per row tracks both species. Drag the gate slider to see the spectrum and its decomposition at that voltage.

gate voltage
Python — NaN-aware smoothing + double-Gaussian tracking 
def nan_gaussian_smooth_2d(z, sigma_gate=1.0, sigma_energy=1.5):
    """Gaussian-smooth a map containing NaNs without letting them spread:
    normalized convolution over the *valid* neighbors only."""
    valid = np.isfinite(z).astype(float)
    filled = np.where(np.isfinite(z), z, 0.0)
    num = gaussian_filter(filled, sigma=(sigma_gate, sigma_energy), mode="nearest")
    den = gaussian_filter(valid,  sigma=(sigma_gate, sigma_energy), mode="nearest")
    out = np.full_like(z, np.nan, dtype=float)
    np.divide(num, den, out=out, where=den > 1e-12)
    return out

def double_gaussian(E, A1, mu1, s1, A2, mu2, s2):
    """Exciton (peak 1) + trion (peak 2)."""
    return (A1 * np.exp(-0.5 * ((E - mu1) / s1) ** 2)
          + A2 * np.exp(-0.5 * ((E - mu2) / s2) ** 2))

# per gate voltage: fixed energy windows keep each Gaussian on its species
p0 = [spec.max(), 1.735, 0.008, spec.max() / 2, 1.700, 0.011]

6Exciton-polariton dispersion

When a microcavity photon and a TMDC exciton couple strongly, the angle-resolved reflectance shows two anticrossing branches. Diagonalizing the 2×2 coupled-oscillator Hamiltonian gives the branch energies and the Hopfield coefficients (photon/exciton content). Play with the detuning and Rabi splitting — the minimum LP–UP gap always equals 2g at zero detuning.

detuning Ec0−EX
Rabi splitting 2g
Python — coupled-oscillator model + Hopfield fractions 
def cavity_dispersion(theta_deg, Ec0, n_eff):
    """Planar-cavity photon dispersion vs external angle."""
    s = np.sin(np.deg2rad(theta_deg)) / n_eff
    return Ec0 / np.sqrt(1.0 - s * s)

def polariton_bands(Ec, EX, twog):
    """LP and UP energies from diagonalizing [[Ec, g], [g, EX]]."""
    d = Ec - EX
    Om = np.sqrt(d * d + twog * twog)
    Emid = 0.5 * (Ec + EX)
    return Emid - 0.5 * Om, Emid + 0.5 * Om

def hopfield_fractions(Ec, EX, twog):
    """|C|^2 (photon) and |X|^2 (exciton) content of the LP branch."""
    d = Ec - EX
    Om = np.maximum(np.sqrt(d * d + twog * twog), 1e-30)
    return 0.5 * (1.0 - d / Om), 0.5 * (1.0 + d / Om)

# fit both extracted branches simultaneously
def residuals(p):
    EX, twog, Ec0, n_eff = p
    Elp, Eup = polariton_bands(cavity_dispersion(theta, Ec0, n_eff), EX, twog)
    return np.concatenate([(Elp - lp)[np.isfinite(lp)],
                           (Eup - up)[np.isfinite(up)]])
res = least_squares(residuals, x0=[1.73, 0.02, 1.72, 2.0],

7The interactive viewer pattern

Every lab viewer shares one layout: a 2D map with a click-to-move crosshair, a spectral linecut (row), a spatial linecut (column), the dynamics at the crosshair, and a frame slider for 3D stacks. class MapViewer:
    def __init__(self, delays, y_um, wl, stack):
        self.im = self.ax_map.imshow(stack[self.i_t], origin="lower",
                                     aspect="auto", cmap="RdBu_r",
                                     vmin=-vmax, vmax=vmax)
        self.ch_h = self.ax_map.axhline(self.cur_y, color="k", lw=0.8, ls="--")
        self.ch_v = self.ax_map.axvline(self.cur_wl, color="k", lw=0.8, ls="--")
        self.slider = Slider(ax_sl, "delay index", 0, delays.size - 1,
                             valinit=self.i_t, valstep=1)
        self.slider.on_changed(self.on_slider)
        self.fig.canvas.mpl_connect("button_press_event", self.on_click)

    def on_click(self, event):
        if event.inaxes is not self.ax_map: return
        self.cur_wl, self.cur_y = event.xdata, event.ydata   # cursor memory
        self.redraw()

    def redraw(self):
        iy = int(np.argmin(np.abs(self.y_um - self.cur_y)))
        iw = int(np.argmin(np.abs(self.wl - self.cur_wl)))
        frame = self.stack[self.i_t]
        self.im.set_data(frame)
        self.spec_line.set_data(self.wl, frame[iy, :])       # row linecut
        self.vert_line.set_data(self.y_um, frame[:, iw])     # column linecut
        self.dyn_line.set_data(self.delays, self.stack[:, iy, iw])
        self.fig.canvas.draw_idle()
